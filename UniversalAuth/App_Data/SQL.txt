CREATE TABLE SYS_USER(
    ID VARCHAR2(50) PRIMARY KEY,
    USERNAME VARCHAR2(20),
    NAME VARCHAR2(10),
    CREATE_DATE DATE,
    VERSION NUMBER
)
CREATE TABLE SYS_MENU(
    ID VARCHAR2(50) PRIMARY KEY,
    PID VARCHAR2(50),
    NAME VARCHAR2(20),
    CONTROLLER VARCHAR2(20),
    ACTION VARCHAR2(20),
	STATUS CHAR(1),--0：菜單，1：權限
    CREATE_DATE DATE,
	VERSION NUMBER
)
CREATE TABLE SYS_PREM(
    ID VARCHAR2(50) PRIMARY KEY,
    URL VARCHAR2(30),
    MENU VARCHAR2(50),
    CREATE_DATE DATE,
    VERSION NUMBER
)
CREATE TABLE SYS_ROLE(
    ID VARCHAR2(50) PRIMARY KEY,
    NAME VARCHAR2(20),
    CREATE_DATE DATE,
    VERSION NUMBER
)
CREATE TABLE R_ROLE_MENU(
    ROLE_ID VARCHAR2(50),
    MENU_ID VARCHAR2(50)    
)
CREATE TABLE R_USER_MENU(
    USER_ID VARCHAR2(50),
    MENU_ID VARCHAR2(50)    
)
CREATE TABLE R_USER_ROLE(
    USER_ID VARCHAR2(50),
    ROLE_ID VARCHAR2(50)    
)
CREATE TABLE SYS_GROUP(
    ID VARCHAR2(50) PRIMARY KEY,
    NAME VARCHAR2(20),
    CREATE_DATE DATE,
    VERSION NUMBER
)
CREATE TABLE R_GROUP_ROLE(
    GROUP_ID VARCHAR2(50),
    ROLE_ID VARCHAR2(50)    
)
CREATE TABLE R_USER_GROUP(
    USER_ID VARCHAR2(50),
    GROUP_ID VARCHAR2(50)    
)
